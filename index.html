function startDarkMusic(){ musicRunning = false; setTimeout(()=>{ if(musicRunning) return; musicRunning = true; const base = 392; // G4 (조금 더 아래, 어둡게) const pattern = [0, -2, -5, -7, -5, -2, 0, -9]; let idx = 0; const stepDur = 640; // 템포 더 느리게 function step(){ if(!musicRunning) return; const semi = pattern[idx % pattern.length]; const f = base * Math.pow(2, semi/12); // 길고 어두운 패드 느낌 playNote(f, 1.2, 0.24, 'sawtooth'); idx++; setTimeout(step, stepDur); } step(); }, 500); }
